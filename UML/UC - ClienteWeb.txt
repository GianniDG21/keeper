    ' -- Casi d'uso del Personale --
    usecase "Gestire Appuntamenti" as UC4
    usecase "Gestire Vetrina Auto" as UC5
    usecase "Gestire Turni Lavoro" as UC6
}

' === Associazioni del Cliente ===
cliente -- UC1

UC1 .> (UC2) : <<include>>

UC3 -- personale

personale -- UC4
